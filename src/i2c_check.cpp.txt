#include <Wire.h>
#include <Arduino.h>
#include <Wire.h>
#include <Adafruit_MPU6050.h>
#include <Adafruit_Sensor.h>

void setup() {
  Wire.begin();
  Serial.begin(9600);
  Serial.println("I2C Scanner");
}
void loop() {
  Serial.println("Test");
  for(byte i=8; i<120; i++) {
    Wire.beginTransmission(i);
    if(Wire.endTransmission()==0) Serial.printf("Found: 0x%02X\n", i);
  }
  delay(500);
}
